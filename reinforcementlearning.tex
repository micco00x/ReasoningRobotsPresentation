\begin{frame}{Q-Learning}
    Q-Learning brief description.
    \begin{equation*}
        Q(S_t, A_t) \leftarrow Q(S_t, A_t) + \alpha \Big[ R_{t+1} +
            \gamma \max_{a} Q(S_{t+1}, a) - Q(S_t, A_t) \Big]
    \end{equation*}
\end{frame}

\begin{frame}{SARSA}
    SARSA brief description.
    \begin{equation*}
        Q(S_t, A_t) \leftarrow Q(S_t, A_t) + \alpha \Big[ R_{t+1} +
            \gamma Q(S_{t+1}, A_{t+1}) - Q(S_t, A_t) \Big]
    \end{equation*}
\end{frame}
